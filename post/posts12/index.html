public class NumOfPrime {
    int numberOfPrime(int n) {
        int result = 0;
        loop1: 
        for(int i=2;i<=n;i++) {

            for(int j=2;j<i;j++) {
                if(i%j==0) {
                    continue loop1;
                }
            }
        result++;   
        }
        // 함수를 완성하세요.

        return result;
    }

    public static void main(String[] args) {
        NumOfPrime prime = new NumOfPrime();
        System.out.println(prime.numberOfPrime(10));
    }

}
