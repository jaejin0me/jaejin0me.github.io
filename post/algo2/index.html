class Collatz {
    public int collatz(long num) {
        int answer = 0;
        int i;
        for(i=0;i<500;i++) {
            if(num==1)
                return answer;          
            if(num%2==0)
                num/=2;         
            else
                num = (num*3)+1;            
            answer++;
        }

        if(num==1)
            return answer;
        else
            return -1;
    }

    // 아래는 테스트로 출력해 보기 위한 코드입니다.
    public static void main(String[] args) {
        Collatz c = new Collatz();
        int ex = 6;
        System.out.println(c.collatz(ex));
    }
}
